namespace = ccrebel_hard

#yearly, on game start
event = {
	id = ccrebel_hard.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		is_difficulty > 1
	}

	immediate = {
		if = {
			limit = {
				NOT = {
					is_difficulty = 3
				}
			}
			every_country = {
				limit = {
					NOT = { has_modifier = ccrebel_hard }
					is_ai = yes
				}
				add_modifier = {
					modifier = ccrebel_hard
					days = -1
				}
			}
			every_country = {
				limit = {
					has_modifier = ccrebel_hard
					is_ai = no
				}
				remove_modifier = ccrebel_hard
			}
		}
		else = {
			every_country = {
				limit = {
					NOT = { has_modifier = ccrebel_insane }
					is_ai = yes
				}
				add_modifier = {
					modifier = ccrebel_insane
					days = -1
				}
			}
			every_country = {
				limit = {
					has_modifier = ccrebel_insane
					is_ai = no
				}
				remove_modifier = ccrebel_insane
			}
		}
	}
}
